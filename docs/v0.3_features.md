    population_size=50,      # 种群大小
    generations=100,         # 进化代数
    mutation_rate=0.1,       # 变异率

    ParameterBound('window', 10, 100, dtype='int'),
    ParameterBound('threshold', 0.1, 1.0, dtype='float'),
    ParameterBound('ma_type', 0, 0, dtype='choice', choices=['sma', 'ema'])

# 执行优化
best = optimizer.optimize(
    strategy_class=MyStrategy,
    parameter_bounds=param_bounds,
    backtest_func=backtest,

    initial_capital=100000

## 📊 完整示例

### 示例 1: 加密货币策略回测

```python
from data.binance_source import BinanceDataSource
from backtest.optimized_engine import optimized_backtest

# 1. 获取数据
source = BinanceDataSource()
data = source.get_klines(
    symbol='BTC/USDT',
    interval='1h',
    start_date='2024-01-01',
    end_date='2024-01-31'
)

# 2. 定义策略
class CryptoStrategy:
    def __init__(self, window=20):
        self.window = window
    
    def generate_signals(self, data):
        import numpy as np
        n = len(data)
        signals = np.zeros(n, dtype=np.int8)
        ma = data['close'].rolling(window=self.window).mean()
        signals[data['close'] > ma] = 1
        signals[data['close'] < ma] = -1
        return signals

# 3. 执行回测
strategy = CryptoStrategy(window=20)
result = optimized_backtest(strategy, data)

print(f"总收益：{result.total_return:.2f}%")
print(f"夏普比率：{result.sharpe_ratio:.2f}")
```
