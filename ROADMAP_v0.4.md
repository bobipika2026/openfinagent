# 伪代码示例
class PaperTradingEngine:
    def __init__(self, strategy, data_feed):
        self.strategy = strategy
        self.data_feed = data_feed
        self.positions = {}
        self.orders = []
        self.cash = 1000000  # 初始资金
        
    def on_tick(self, tick_data):
        # 推送实时数据给策略
        signals = self.strategy.on_tick(tick_data)
        
        # 模拟订单执行
        for signal in signals:
            order = self.create_order(signal)
            self.execute_order(order)  # 模拟成交
            
        # 更新持仓和收益
        self.update_portfolio()

```python
# 数据源统一接口
class DataFeed(ABC):
    @abstractmethod
    def get_history(self, symbol, start, end, frequency):
        pass
    
    @abstractmethod
    def get_realtime(self, symbol):
        pass
    
    @abstractmethod
    def check_quality(self):
        pass

# Yahoo Finance 实现
class YahooFinanceFeed(DataFeed):
    def __init__(self):
        import yfinance as yf
        self.yf = yf
        
    def get_history(self, symbol, start, end, frequency):
        ticker = self.yf.Ticker(symbol)
        return ticker.history(start=start, end=end, interval=frequency)

# 数据源管理器
class DataManager:
    def __init__(self):
        self.feeds = {
            'tushare': TushareFeed(),
            'yahoo': YahooFinanceFeed(),
            'alphavantage': AlphaVantageFeed()
        }
        self.primary = 'tushare'
        
    def get_data(self, symbol, **kwargs):
        # 尝试主数据源，失败则降级
        try:
            return self.feeds[self.primary].get_history(symbol, **kwargs)
        except Exception:
            # 自动切换到备用数据源
            for feed_name in ['yahoo', 'alphavantage']:
                try:
                    return self.feeds[feed_name].get_history(symbol, **kwargs)
                except Exception:
                    continue
            raise Exception("所有数据源均不可用")
```
